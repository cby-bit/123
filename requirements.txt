from flask import Flask, redirect, request
import datetime
import csv
import os

app = Flask(__name__)
CSV_FILE = "clicks.csv"

@app.route("/track")
def track():
    target = "https://www.youtube.com/"  # ← 改成你想跳轉的網址
    click_time = datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    user_ip = request.remote_addr

    # 將點擊記錄存入 CSV
    file_exists = os.path.isfile(CSV_FILE)
    with open(CSV_FILE, mode="a", newline="", encoding="utf-8") as f:
        writer = csv.writer(f)
        if not file_exists:
            writer.writerow(["time", "ip", "target"])
        writer.writerow([click_time, user_ip, target])

    print(f"{click_time} - 點擊來自 {user_ip} - 目標：{target}")
    return redirect(target)

# ⚠️ 不要寫 app.run()
